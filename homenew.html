function npv(x, m, f=3) {
    let p = 1/(Math.pow(1+m,x))
    return p.toFixed(f)
}
function doTable() {
    const L = [10, 12, 14, 15, 16]
    let a = L.map((m) => 'm='+m)
    let str = header(a)
    for (let x=0; x<=10; x++) { 
       let a = L.map((m) => npv(x, m))
       str += oneLine(a, x)
    }
    return str
}
